-- 创建数据表空间 LCMS_BLANK_DATA
create tablespace LCMS_BLANK_DATA datafile 'D:\oracle11g\oradata\LCMS_BLANK_DATA.dbf' size 100m
AUTOEXTEND ON NEXT 5M MAXSIZE UNLIMITED
LOGGING
PERMANENT
EXTENT MANAGEMENT LOCAL AUTOALLOCATE
BLOCKSIZE 8K
SEGMENT SPACE MANAGEMENT MANUAL;

-- 创建索引表空间 LCMS_BLANK_INDEX
create tablespace LCMS_INDEX datafile 'D:\oracle11g\oradata\LCMS_BLANK_INDEX.dbf' size 100m
AUTOEXTEND ON NEXT 5M MAXSIZE UNLIMITED
LOGGING
PERMANENT
EXTENT MANAGEMENT LOCAL AUTOALLOCATE
BLOCKSIZE 8K
SEGMENT SPACE MANAGEMENT MANUAL;


-- 创建用户密码: LCMS_BLANK/LCMS_BLANK
create user  LCMS_BLANK identified by LCMS_BLANK default tablespace LCMS_BLANK_DATA;
